\begin{frame}{\texttt{> 02 ::: differential privacy in 30 seconds}}

\begin{block}{\ttfamily def: $\eps$-differential privacy}
A randomized mechanism $\mathcal{M}$ is $\eps$-DP if for any two neighbouring databases $D \sim D'$ (differ by one row) and any output $S$:
\[
\Pr[\mathcal{M}(D) \in S] \;\leq\; e^{\eps} \cdot \Pr[\mathcal{M}(D') \in S]
\]
\end{block}

\vspace{0.4em}
{\color{dim}\ttfamily // in practice: Laplace mechanism}
\[
\widetilde{f}(D) \;=\; f(D) + \eta, \qquad \eta \sim \text{Lap}(\Delta f / \eps)
\]

\vspace{0.2em}
{\color{dim}\ttfamily // budget composition: $\eps$ accumulates over queries.}

\vspace{0.8em}
\begin{alertblock}{\ttfamily the engineering question}
A finite budget $\eps_\text{total}$ is consumed as the workload runs. \\[0.2em]
\hi{HOW does it degrade?} Predicting this \emph{before deployment} is what this project does.
\end{alertblock}

\end{frame}

\begin{frame}{\texttt{> 06 ::: the model in one slide}}

{\color{dim}\ttfamily // setup: workload of $k$ queries drawn i.i.d.\ from $\{p_i\}_{i=1}^m$.}

\vspace{0.3em}
\begin{block}{\ttfamily prop 1 -- expected unique queries}
\[
\E[u_k] \;=\; \sum_{i=1}^m \bigl[1 - (1-p_i)^k\bigr]
\]
{\color{dim}\small (coupon-collector / occupancy on a non-uniform distribution)}
\end{block}

\begin{block}{\ttfamily prop 2 -- workload-aware budget}
\[
\E[\eps_\text{wa}] \;=\; \eps_q \cdot \E[u_k]
\qquad
S(k) \;=\; 1 - \tfrac{1}{k}\sum_i [1 - (1-p_i)^k]
\]
\end{block}

\vspace{0.3em}
{\color{neon}\ttfamily two limits:}
\begin{itemize}
  \item $\alpha \to \infty$ \;(perfect repetition, $p_1 = 1$): $\;S(k) = 1 - 1/k$ \hfill \hi{// the milestone's toy formula}
  \item $\alpha \to 0$ \;(uniform, $m \to \infty$): $\;S(k) \to 0$ \hfill \hi{// naive composition recovered}
\end{itemize}

\end{frame}

\begin{frame}{\texttt{> 12 ::: predictive allocator -- the new mechanism}}

{\color{dim}\ttfamily // use the model's $\E[u_k]$ estimate at runtime to choose $\eps_q$:}

\vspace{0.4em}
\begin{enumerate}
  \item observe template frequencies online
  \item estimate $\hat U = \E[u_k \mid \hat p]$ from Proposition 1
  \item allocate $\eps_q^\text{pred} = B / \hat U$
\end{enumerate}

\vspace{0.5em}
\begin{exampleblock}{\ttfamily novel because}
PINQ / PrivateSQL / Chorus / DOP-SQL all fix $\eps_q$ \emph{offline}. \\
This is the \hi{first DP-SQL mechanism} that adapts $\eps_q$ at runtime from a learned model.
\end{exampleblock}

\begin{frame}{\texttt{> 13 ::: ai/nlp bonus -- semantic cache}}

{\color{dim}\ttfamily // idea: two queries with similar ASTs might share a cache entry.}

\vspace{0.3em}
{\color{neon}\ttfamily // two similarity signals:}
\begin{itemize}
  \item \hi{Tree Kernel} (Collins--Duffy 2001) -- shared subtree count, no training
  \item \hi{AST embedding} (sentence-transformers / CodeBERT) -- cosine on dense vector
\end{itemize}

\vspace{0.5em}
\begin{alertblock}{\ttfamily honest negative result}
The semantic L2 cache fails in \hi{both} directions:
\begin{itemize}
  \item \hilo{false positives:} reuses cached noisy answer on queries with \emph{different} true answers $\Rightarrow$ wrong by thousands of count units
  \item \hilo{false negatives:} misses textbook equivalences \\
  (commutative AND, integer comparators, redundant predicates)
\end{itemize}
\vspace{0.2em}
\textbf{\hi{semantic similarity $\neq$ formal equivalence.}} \\
``AI-powered DP cache'' needs a symbolic equivalence prover on top.
\end{alertblock}

\end{frame}

\begin{frame}{\texttt{> 16 ::: what we found -- four takeaways}}

\begin{enumerate}
  \item \hi{Workload-aware budget is structural, not a cache trick.} \\
  \dimt{$\E[\eps_\text{wa}] = \eps_q \cdot \E[u_k]$ predicts within 3\% on the main grid, \emph{exact} on the mixed workload, same at SF=1 and SF=10.}

  \vspace{0.5em}
  \item \hi{The $1-1/k$ toy rule is the $\alpha \to \infty$ corner case.} \\
  \dimt{The model recovers it \emph{and} the entire interior up to naive composition.}

  \vspace{0.5em}
  \item \hi{Cumulative-$\eps$ overestimates real attack success.} \\
  \dimt{Shadow MIA on W1--W4 stays at AUC 0.15--0.58 vs.\ a 1.0 cumulative bound -- budget is over-provisioned in practice.}

  \vspace{0.5em}
  \item \hi{Semantic similarity $\neq$ equivalence.} \\
  \dimt{Tree-kernel + embedding admits false positives \emph{and} misses textbook equivalences. ``AI-powered DP'' alone is not DP-safe.}
\end{enumerate}

\end{frame}
