\begin{frame}{The Problem: Aggregate Queries Are Not Safe}

\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}
\textbf{Intuition (wrong):}
``Aggregate queries return totals, not individual rows -- they must be safe.''

\vspace{0.3cm}
\textbf{Reality:} Two overlapping COUNTs isolate one person.

\vspace{0.3cm}
\begin{lstlisting}
-- 12 patients with HIV
SELECT COUNT(*) FROM patients
WHERE zip='06897' AND disease='HIV';

-- 11 patients with HIV, age <> 43
SELECT COUNT(*) FROM patients
WHERE zip='06897' AND disease='HIV'
  AND age <> 43;
\end{lstlisting}

\vspace{0.2cm}
$\Rightarrow$ The 43-year-old in 06897 has HIV.
\end{column}

\begin{column}{0.45\textwidth}
\centering

\begin{frame}{Differential Privacy in 30 Seconds}

\begin{block}{Definition (\eps-DP)}
A randomized mechanism $\mathcal{M}$ is \eps-DP if for any two neighbouring databases $D \sim D'$ (differ by one row) and any output $S$:
\[
\Pr[\mathcal{M}(D) \in S] \leq e^{\eps} \cdot \Pr[\mathcal{M}(D') \in S]
\]
\end{block}

\vspace{0.3cm}
\textbf{In practice (Laplace mechanism):}
\[
\tilde f(D) = f(D) + \eta, \quad \eta \sim \text{Lap}(\Delta f / \eps)
\]

\vspace{0.2cm}
\textbf{Budget composition:} \eps\ accumulates over queries.

\vspace{0.5cm}
\begin{alertblock}{The Engineering Question}
A finite budget $\eps_{\text{total}}$ is consumed as the workload runs. \\
\textbf{How does it degrade?} Predicting this \emph{before} deployment is what this project answers.
\end{alertblock}

\end{frame}

\begin{frame}{The Model in One Slide}

\textbf{Setup:} workload of $k$ queries drawn i.i.d.\ from a template distribution $\{p_i\}_{i=1}^m$.

\vspace{0.3cm}
\begin{block}{Proposition 1 -- Expected unique queries}
\[
\E[u_k] = \sum_{i=1}^m \bigl[1 - (1-p_i)^k\bigr]
\]
Coupon-collector / occupancy on a non-uniform distribution.
\end{block}

\begin{block}{Proposition 2 -- Workload-aware budget}
\[
\E[\eps_{\text{wa}}] = \eps_q \cdot \E[u_k] \qquad
S(k) = 1 - \tfrac{1}{k}\sum_i \bigl[1 - (1-p_i)^k\bigr]
\]
\end{block}

\textbf{Two limits:}
\begin{itemize}
  \item $\alpha \to \infty$ (perfect repetition, $p_1 = 1$): $\;S(k) = 1 - 1/k$ \hfill \textcolor{ok}{(the milestone's toy formula)}
  \item $\alpha \to 0$ (uniform, $m \to \infty$): $\;S(k) \to 0$ \hfill \textcolor{ok}{(naive composition recovered)}
\end{itemize}

\end{frame}

\begin{frame}{Putting the Model to Work: Predictive Allocator}

\textbf{The new mechanism:} Use the model's $\E[u_k]$ estimate at runtime to choose $\eps_q$.

\vspace{0.3cm}
\begin{enumerate}
  \item Observe template frequencies online
  \item After warmup, estimate $\hat U = \E[u_k \mid \hat p]$ from Proposition~1
  \item Allocate $\eps_q^{\text{pred}} = B / \hat U$
\end{enumerate}

\vspace{0.3cm}
\textbf{Novel because:} PINQ, PrivateSQL, Chorus, DOP-SQL all fix $\eps_q$ offline. \\
\hspace{2em} This is the first DP-SQL mechanism that adapts $\eps_q$ at runtime from a learned model.

\begin{frame}{AI / NLP Bonus: Semantic Cache via Tree Kernel + AST Embedding}

\textbf{Idea:} two queries with the same AST (modulo literals) might be safe to share a cache entry.

\vspace{0.3cm}
\textbf{Two similarity signals:}
\begin{itemize}
  \item \textbf{Tree Kernel} (Collins--Duffy): count shared subtrees -- deterministic, no training
  \item \textbf{AST Embedding} (sentence-transformers / CodeBERT): dense vector, cosine similarity
\end{itemize}

\vspace{0.3cm}
\begin{alertblock}{Honest negative result}
The semantic L2 cache fails in \textbf{both directions}:
\begin{itemize}
  \item \textbf{False positives:} reuses cached noisy answer on queries with different true answers $\Rightarrow$ wrong by thousands
  \item \textbf{False negatives:} misses textbook equivalences (commutative AND, integer comparators)
\end{itemize}
\textbf{Conclusion:} Semantic similarity $\neq$ formal equivalence. \\
``AI-powered DP cache'' needs a \emph{symbolic equivalence prover} on top.
\end{alertblock}

\end{frame}

\begin{frame}{What We Found -- Four Things Worth Stating}

\begin{enumerate}
\item \textbf{Workload-aware budget is a structural property, not a cache trick.}\\[0.2cm]
$\E[\eps_{\text{wa}}] = \eps_q \cdot \E[u_k]$ predicts within 3\% on the main grid, \emph{exact} on the mixed W1+W4 workload, and the same at SF=1 and SF=10.\\

\item \textbf{The toy $1-1/k$ rule is the $\alpha \to \infty$ corner case.}\\[0.2cm]
Not a standalone claim -- the model recovers it and the entire Zipf interior between it and naive composition.\\

\item \textbf{Cumulative-$\eps$ overestimates real attack success.}\\[0.2cm]
Shadow MIA on W1--W4 stays at AUC 0.15--0.58 vs.\ a 1.0 cumulative bound. Budget tends to be \emph{over}-provisioned in practice.\\

\item \textbf{Semantic similarity is not equivalence.}\\[0.2cm]
Tree-kernel + embedding admits dangerous false positives and misses textbook equivalences. ``AI-powered DP'' alone is not DP-safe.
\end{enumerate}

\end{frame}

\begin{frame}{Honest Limitations \& Future Work}

\textbf{What this project is not:}
\begin{itemize}
  \item \emph{Not} a head-to-head comparison against real DP-SQL systems (PINQ, PrivateSQL, Chorus, DOP-SQL re-implemented from scratch is multi-month work)
  \item \emph{Not} a real-workload trace -- Zipf parametrization is a controlled sweep
  \item \emph{Not} a new privacy mechanism in the cryptographic sense
\end{itemize}

\vspace{0.4cm}
\textbf{Concrete next steps (paper \S11):}
\begin{itemize}
  \item Cache re-release on budget surplus (closes predictive's main gap)
  \item R\'enyi DP / zCDP composition (tighter bound)
  \item Burstiness via renewal processes
  \item Real workload traces (Snowflake / Redshift anonymized)
  \item Joint budget--leakage bound that beats cumulative-$\eps$
  \item Equivalence-checked semantic cache (similarity + symbolic prover)
  \item Multi-table / JOIN via residual sensitivity or R2T
\end{itemize}
